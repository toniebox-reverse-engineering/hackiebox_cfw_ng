152 int spi_Read(Fd_t fd, unsigned char *pBuff, int len)
153 {
154  int read_size = 0;
155  if(fd < 0)
156  return -1;
157 
158  while (len>0)
159  {
160  if( len < MAX_DMA_RECV_TRANSACTION_SIZE)
161  {
162  spi_read((void *)fd, (char *)&pBuff[read_size], len);
163  read_size += len;
164  len = 0;
165  }
166  else
167  {
168  spi_read((void *)fd, (char *)&pBuff[read_size],
169  MAX_DMA_RECV_TRANSACTION_SIZE);
170  read_size += MAX_DMA_RECV_TRANSACTION_SIZE;
171  len -= MAX_DMA_RECV_TRANSACTION_SIZE;
172  }
173  }
174 
175  return read_size;
176 }

296 void NwpPowerOnPreamble(void)
297 {
298 #ifndef CC3200_ES_1_2_1
299 
300 #define MAX_RETRY_COUNT 1000
301  unsigned int sl_stop_ind, apps_int_sts_raw, nwp_lpds_wake_cfg;
302  unsigned int retry_count;
303  /* Perform the sl_stop equivalent to ensure network services
304  are turned off if active */
305  HWREG(0x400F70B8) = 1; /* APPs to NWP interrupt */
306  UtilsDelay(800000/5);
307 
308  retry_count = 0;
309  nwp_lpds_wake_cfg = HWREG(0x4402D404);
310  sl_stop_ind = HWREG(0x4402E16C);
311 
312  if((nwp_lpds_wake_cfg != 0x20) && /* Check for NWP POR condition */
313  !(sl_stop_ind & 0x2)) /* Check if sl_stop was executed */
314  {
315  /* Loop until APPs->NWP interrupt is cleared or timeout */
316  while(retry_count < MAX_RETRY_COUNT)
317  {
318  apps_int_sts_raw = HWREG(0x400F70C0);
319  if(apps_int_sts_raw & 0x1)
320  {
321  UtilsDelay(800000/5);
322  retry_count++;
323  }
324  else
325  {
326  break;
327  }
328  }
329  }
330  HWREG(0x400F70B0) = 1; /* Clear APPs to NWP interrupt */
331  UtilsDelay(800000/5);
332 
333  /* Stop the networking services */
334  NwpPowerOff();
335 #endif
336 }
337 #endif

379 void NwpPowerOff(void)
380 {
381  //Must delay 300 usec to enable the NWP to finish all sl_stop activities
382  UtilsDelay(300*80/3);
383 
384  //Mask Host Interrupt
385  NwpMaskInterrupt();
386 
387  //Switch to PFM Mode
388  HWREG(0x4402F024) &= 0xF7FFFFFF;
389 #ifdef CC3200_ES_1_2_1
390  //Reset NWP
391  HWREG(APPS_SOFT_RESET_REG) |= 4;
392  //WLAN PD OFF
393  HWREG(OCP_SHARED_MAC_RESET_REG) |= 0xC00;
394 #else
395  //sl_stop eco for PG1.32 devices
396  HWREG(0x4402E16C) |= 0x2;
397 
398  UtilsDelay(800000);
399 #endif
400 }

411 void NwpPowerOff_WithNwpLpdsPoll(void)
412 {
413  //Must delay 300 usec to enable the NWP to finish all sl_stop activities
414  UtilsDelay(300*80/3);
415 
416  //Mask Host Interrupt
417  NwpMaskInterrupt();
418 
419 #ifdef CC3200_ES_1_2_1
420  //Reset NWP
421  HWREG(APPS_SOFT_RESET_REG) |= 4;
422  //WLAN PD OFF
423  HWREG(OCP_SHARED_MAC_RESET_REG) |= 0xC00;
424 #else
425 
426  // Dynamic delay loop to make sure the NWP has entered low power mode
427  {
428  // Wait for NWP to enter LPDS with 100 mSec timeout
429  unsigned int NwpActive;
430  long TimeoutUsec;
431 
432  TimeoutUsec = 100000;
433 
434  do
435  {
436  NwpActive = ((HWREG(0x4402DC48) & 0xF00) == 0x300);
437  TimeoutUsec -= 10;
438  UtilsDelay(800/6);
439  }while (NwpActive && (TimeoutUsec > 0));
440  }
441 
442  // Switch to PFM Mode
443  // Note: make sure the switch to PFM Mode is done just before killing the
444  // network processor
445  HWREG(0x4402F024) &= 0xF7FFFFFF;
446 
447  //Killing the NWP
448  HWREG(0x4402E16C) |= 0x2;
449 
450  UtilsDelay(800000);
451 #endif
452 }
