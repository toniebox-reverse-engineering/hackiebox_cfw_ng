61 _SlNonOsRetVal_t _SlNonOsSemGet(_SlNonOsSemObj_t* pSyncObj, _SlNonOsSemObj_t WaitValue, _SlNonOsSemObj_t SetValue, _SlNonOsTime_t Timeout)
62 {
63 #if (!defined (SL_TINY)) && (defined(sl_GetTimestamp))
64  _SlTimeoutParams_t TimeoutInfo={0};
65 #endif
66 
67  /* If timeout 0 configured, just detect the value and return */
68  if ((Timeout ==0) && (WaitValue == *((volatile _u8 *)pSyncObj)) )
69  {
70  *pSyncObj = SetValue;
71  return NONOS_RET_OK;
72  }
73 
74 #if (!defined (SL_TINY)) && (defined(sl_GetTimestamp))
75  if ((Timeout != NONOS_WAIT_FOREVER) && (Timeout != NONOS_NO_WAIT))
76  {
77  _SlDrvStartMeasureTimeout(&TimeoutInfo, Timeout);
78  }
79 #endif
80 
81 #ifdef _SlSyncWaitLoopCallback
82  _SlNonOsTime_t timeOutRequest = Timeout;
83 #endif
84  while (Timeout>0)
85  {
86  if (WaitValue == *((volatile _u8 *)pSyncObj))
87  {
88  *pSyncObj = SetValue;
89  break;
90  }
91 #if (!defined (sl_GetTimestamp)) || (defined (SL_TINY_EXT))
92  if (Timeout != NONOS_WAIT_FOREVER)
93  {
94  Timeout--;
95  }
96 #else
97  if ((Timeout != NONOS_WAIT_FOREVER) && (Timeout != NONOS_NO_WAIT))
98  {
99  if (_SlDrvIsTimeoutExpired(&TimeoutInfo))
100  {
101  return (_SlNonOsRetVal_t)NONOS_RET_ERR;
102  }
103 
104  }
105  #endif
106 
107  /* If we are in cmd context and waiting for its cmd response
108  * do not handle spawn async events as the global lock was already taken */
109  if (FALSE == g_pCB->IsCmdRespWaited)
110  {
111  (void)_SlNonOsMainLoopTask();
112  }
113 #ifdef _SlSyncWaitLoopCallback
114  if( (__NON_OS_SYNC_OBJ_SIGNAL_VALUE == WaitValue) && (timeOutRequest != NONOS_NO_WAIT) )
115  {
116  if (WaitValue == *((volatile _u8 *)pSyncObj))
117  {
118  *pSyncObj = SetValue;
119  break;
120  }
121  _SlSyncWaitLoopCallback();
122  }
123 #endif
124  }
125 
126  if (0 == Timeout)
127  {
128  return NONOS_RET_ERR;
129  }
130  else
131  {
132  return NONOS_RET_OK;
133  }
134 }

165 _SlNonOsRetVal_t _SlNonOsMainLoopTask(void)
166 {
167  _i8 i=0;
168  void* pValue;
169 
170 #ifndef SL_TINY_EXT
171  for (i=0 ; i<NONOS_MAX_SPAWN_ENTRIES ; i++)
172 #endif
173  {
174  _SlNonOsSpawnEntry_t* pE = &g__SlNonOsCB.SpawnEntries[i];
175 
176 
177  if (pE->IsAllocated == TRUE)
178  {
179  _SlSpawnEntryFunc_t pF = pE->pEntry;
180  pValue = pE->pValue;
181 
182 
183  /* Clear the entry */
184  pE->pEntry = NULL;
185  pE->pValue = NULL;
186  pE->IsAllocated = FALSE;
187 
188  /* execute the spawn function */
189  pF(pValue);
190  }
191  }
192 
193  return NONOS_RET_OK;
194 }
