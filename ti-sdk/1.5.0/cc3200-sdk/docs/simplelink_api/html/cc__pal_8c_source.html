155 int spi_Read_CPU(unsigned char *pBuff, int len)
156 {
157  unsigned long ulCnt;
158  unsigned long ulStatusReg;
159  unsigned long *ulDataIn;
160  unsigned long ulTxReg;
161  unsigned long ulRxReg;
162 
163  MAP_SPICSEnable(LSPI_BASE);
164 
165  //
166  // Initialize local variable.
167  //
168  ulDataIn = (unsigned long *)pBuff;
169  ulCnt = (len + 3) >> 2;
170  ulStatusReg = LSPI_BASE+MCSPI_O_CH0STAT;
171  ulTxReg = LSPI_BASE + MCSPI_O_TX0;
172  ulRxReg = LSPI_BASE + MCSPI_O_RX0;
173 
174  //
175  // Reading loop
176  //
177  while(ulCnt--)
178  {
179  while(!( HWREG(ulStatusReg)& MCSPI_CH0STAT_TXS ));
180  HWREG(ulTxReg) = 0xFFFFFFFF;
181  while(!( HWREG(ulStatusReg)& MCSPI_CH0STAT_RXS ));
182  *ulDataIn = HWREG(ulRxReg);
183  ulDataIn++;
184  }
185 
186  MAP_SPICSDisable(LSPI_BASE);
187 
188  return len;
189 }

209 int spi_Write_CPU(unsigned char *pBuff, int len)
210 {
211 
212 
213  unsigned long ulCnt;
214  unsigned long ulStatusReg;
215  unsigned long *ulDataOut;
216  unsigned long ulDataIn;
217  unsigned long ulTxReg;
218  unsigned long ulRxReg;
219 
220 
221  MAP_SPICSEnable(LSPI_BASE);
222 
223  //
224  // Initialize local variable.
225  //
226  ulDataOut = (unsigned long *)pBuff;
227  ulCnt = (len +3 ) >> 2;
228  ulStatusReg = LSPI_BASE+MCSPI_O_CH0STAT;
229  ulTxReg = LSPI_BASE + MCSPI_O_TX0;
230  ulRxReg = LSPI_BASE + MCSPI_O_RX0;
231 
232  //
233  // Writing Loop
234  //
235  while(ulCnt--)
236  {
237  while(!( HWREG(ulStatusReg)& MCSPI_CH0STAT_TXS ));
238  HWREG(ulTxReg) = *ulDataOut;
239  while(!( HWREG(ulStatusReg)& MCSPI_CH0STAT_RXS ));
240  ulDataIn = HWREG(ulRxReg);
241  ulDataOut++;
242  }
243 
244  MAP_SPICSDisable(LSPI_BASE);
245 
246  UNUSED(ulDataIn);
247  return len;
248

387 Fd_t spi_Open(char *ifName, unsigned long flags)
388 {
389  unsigned long ulBase;
390  unsigned long ulSpiBitRate = 0;
391  tROMVersion* pRomVersion = (tROMVersion *)(ROM_VERSION_ADDR);
392 
393 
394  //NWP master interface
395  ulBase = LSPI_BASE;
396 
397  //Enable MCSPIA2
398  MAP_PRCMPeripheralClkEnable(PRCM_LSPI,PRCM_RUN_MODE_CLK|PRCM_SLP_MODE_CLK);
399 
400  //Disable Chip Select
401  MAP_SPICSDisable(ulBase);
402 
403  //Disable SPI Channel
404  MAP_SPIDisable(ulBase);
405 
406  // Reset SPI
407  MAP_SPIReset(ulBase);
408 
409  //
410  // Configure SPI interface
411  //
412 
413  if(pRomVersion->ucMinorVerNum == ROM_VER_PG1_21 )
414  {
415  ulSpiBitRate = SPI_RATE_13M;
416  }
417  else if(pRomVersion->ucMinorVerNum == ROM_VER_PG1_32)
418  {
419  ulSpiBitRate = SPI_RATE_13M;
420  }
421  else if(pRomVersion->ucMinorVerNum >= ROM_VER_PG1_33)
422  {
423  ulSpiBitRate = SPI_RATE_20M;
424  }
425 
426  MAP_SPIConfigSetExpClk(ulBase,MAP_PRCMPeripheralClockGet(PRCM_LSPI),
427  ulSpiBitRate,SPI_MODE_MASTER,SPI_SUB_MODE_0,
428  (SPI_SW_CTRL_CS |
429  SPI_4PIN_MODE |
430  SPI_TURBO_OFF |
431  SPI_CS_ACTIVEHIGH |
432  SPI_WL_32));
433 
434  if(MAP_PRCMPeripheralStatusGet(PRCM_UDMA))
435  {
436  g_ucDMAEnabled = (HWREG(UDMA_BASE + UDMA_O_CTLBASE) != 0x0) ? 1 : 0;
437  }
438  else
439  {
440  g_ucDMAEnabled = 0;
441  }
442  #ifdef SL_CPU_MODE
443  g_ucDMAEnabled = 0;
444  #endif
445  if(g_ucDMAEnabled)
446  {
447  memset(g_ucDinDout,0xFF,sizeof(g_ucDinDout));
448 
449  // Set DMA channel
450  cc_UDMAChannelSelect(UDMA_CH12_LSPI_RX);
451  cc_UDMAChannelSelect(UDMA_CH13_LSPI_TX);
452 
453 
454  MAP_SPIFIFOEnable(ulBase,SPI_RX_FIFO);
455  MAP_SPIFIFOEnable(ulBase,SPI_TX_FIFO);
456  MAP_SPIDmaEnable(ulBase,SPI_RX_DMA);
457  MAP_SPIDmaEnable(ulBase,SPI_TX_DMA);
458 
459  MAP_SPIFIFOLevelSet(ulBase,1,1);
460  #if defined(SL_PLATFORM_MULTI_THREADED)
461  osi_InterruptRegister(INT_LSPI, (P_OSI_INTR_ENTRY)DmaSpiSwIntHandler,INT_PRIORITY_LVL_1);
462  MAP_SPIIntEnable(ulBase,SPI_INT_EOW);
463 
464 
465  osi_MsgQCreate(&DMAMsgQ,"DMAQueue",sizeof(int),1);
466 
467  #else
468 
469  MAP_IntRegister(INT_LSPI,(void(*)(void))DmaSpiSwIntHandler);
470  MAP_IntPrioritySet(INT_LSPI, INT_PRIORITY_LVL_1);
471  MAP_IntEnable(INT_LSPI);
472 
473  MAP_SPIIntEnable(ulBase,SPI_INT_EOW);
474 
475 
476  g_cDummy = 0x0;
477  #endif
478 
479  }
480  MAP_SPIEnable(ulBase);
481 
482  g_SpiFd = 1;
483  return g_SpiFd;
484

548 int spi_Read(Fd_t fd, unsigned char *pBuff, int len)
549 {
550  int read_size = 0;
551  unsigned int pBuffAddr = (unsigned int)pBuff;
552  if(fd!=1 || g_SpiFd!=1)
553  return -1;
554 
555  if(len>DMA_BUFF_SIZE_MIN && g_ucDMAEnabled && ((pBuffAddr % 4) == 0))
556  {
557 
558 #if defined(SL_PLATFORM_MULTI_THREADED)
559  char temp[4];
560 #endif
561  while (len>0)
562  {
563  if( len < MAX_DMA_RECV_TRANSACTION_SIZE)
564  {
565  SetupDMAReceive(&pBuff[read_size],len);
566  MAP_SPICSEnable(LSPI_BASE);
567 #if defined(SL_PLATFORM_MULTI_THREADED)
568  osi_MsgQRead(&DMAMsgQ,temp,OSI_WAIT_FOREVER);
569 #else
570  while(g_cDummy != 0x1);
571  g_cDummy = 0x0;
572 #endif
573  read_size += len;
574  len = 0;
575  }
576  else
577  {
578  SetupDMAReceive(&pBuff[read_size],MAX_DMA_RECV_TRANSACTION_SIZE);
579  MAP_SPICSEnable(LSPI_BASE);
580 #if defined(SL_PLATFORM_MULTI_THREADED)
581  osi_MsgQRead(&DMAMsgQ,temp,OSI_WAIT_FOREVER);
582 #else
583  while(g_cDummy != 0x1);
584  g_cDummy = 0x0;
585 #endif
586  read_size += MAX_DMA_RECV_TRANSACTION_SIZE;
587 
588  len -= MAX_DMA_RECV_TRANSACTION_SIZE;
589  }
590  }
591  }
592  else
593  {
594  MAP_SPIWordCountSet(LSPI_BASE,0);
595  read_size += spi_Read_CPU(pBuff,len);
596  }
597  return read_size;
598 }

738 void NwpPowerOnPreamble(void)
739 {
740 #ifndef CC3200_ES_1_2_1
741 
742 #define MAX_RETRY_COUNT 1000
743  unsigned int sl_stop_ind, apps_int_sts_raw, nwp_lpds_wake_cfg;
744  unsigned int retry_count;
745  /* Perform the sl_stop equivalent to ensure network services
746  are turned off if active */
747  HWREG(0x400F70B8) = 1; /* APPs to NWP interrupt */
748  UtilsDelay(800000/5);
749 
750  retry_count = 0;
751  nwp_lpds_wake_cfg = HWREG(0x4402D404);
752  sl_stop_ind = HWREG(0x4402E16C);
753 
754  if((nwp_lpds_wake_cfg != 0x20) && /* Check for NWP POR condition */
755  !(sl_stop_ind & 0x2)) /* Check if sl_stop was executed */
756  {
757  /* Loop until APPs->NWP interrupt is cleared or timeout */
758  while(retry_count < MAX_RETRY_COUNT)
759  {
760  apps_int_sts_raw = HWREG(0x400F70C0);
761  if(apps_int_sts_raw & 0x1)
762  {
763  UtilsDelay(800000/5);
764  retry_count++;
765  }
766  else
767  {
768  break;
769  }
770  }
771  }
772  HWREG(0x400F70B0) = 1; /* Clear APPs to NWP interrupt */
773  UtilsDelay(800000/5);
774 
775  /* Stop the networking services */
776  NwpPowerOff();
777 #endif
778 }
779 #endif

844  HWREG(0x4402E16C) |= 0x2;

846  UtilsDelay(800000);
847 #endif
848 }

859 void NwpPowerOff_WithNwpLpdsPoll(void)
860 {
861  //Must delay 300 usec to enable the NWP to finish all sl_stop activities
862  UtilsDelay(300*80/3);
863 
864  //Mask Host Interrupt
865  NwpMaskInterrupt();
866 
867 #ifdef CC3200_ES_1_2_1
868  //Reset NWP
869  HWREG(APPS_SOFT_RESET_REG) |= 4;
870  //WLAN PD OFF
871  HWREG(OCP_SHARED_MAC_RESET_REG) |= 0xC00;
872 #else
873 
874  // Dynamic delay loop to make sure the NWP has entered low power mode
875  {
876  // Wait for NWP to enter LPDS with 100 mSec timeout
877  unsigned int NwpActive;
878  long TimeoutUsec;
879 
880  TimeoutUsec = 100000;
881 
882  do
883  {
884  NwpActive = ((HWREG(0x4402DC48) & 0xF00) == 0x300);
885  TimeoutUsec -= 10;
886  UtilsDelay(800/6);
887  }while (NwpActive && (TimeoutUsec > 0));
888  }
889 
890  // Switch to PFM Mode
891  // Note: make sure the switch to PFM Mode is done just before killing the
892  // network processor
893  HWREG(0x4402F024) &= 0xF7FFFFFF;
894 
895  //Killing the NWP
896  HWREG(0x4402E16C) |= 0x2;
897 
898  UtilsDelay(800000);
899 #endif
