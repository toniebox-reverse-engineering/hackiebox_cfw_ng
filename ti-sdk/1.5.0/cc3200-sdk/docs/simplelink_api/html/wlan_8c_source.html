73 _i16 sl_WlanConnect(const _i8* pName,const _i16 NameLen,const _u8 *pMacAddr,const SlSecParams_t* pSecParams ,const SlSecParamsExt_t* pSecExtParams)
74 {
75  _SlWlanConnectMsg_u Msg;
76  _SlCmdCtrl_t CmdCtrl = {0,0,0};
77 
78  _SlDrvMemZero(&Msg, (_u16)sizeof(_SlWlanConnectMsg_u));
79 
80  /* verify no erorr handling in progress. if in progress than
81  ignore the API execution and return immediately with an error */
82  VERIFY_NO_ERROR_HANDLING_IN_PROGRESS();
83 
84  CmdCtrl.TxDescLen = 0;/* init */
85  CmdCtrl.RxDescLen = (_SlArgSize_t)sizeof(_BasicResponse_t);
86 
87  /* verify SSID length */
88  VERIFY_PROTOCOL(NameLen <= MAX_SSID_LEN);
89  /* verify SSID is not NULL */
90  if( NULL == pName )
91  {
92  return SL_INVALPARAM;
93  }
94  /* update SSID length */
95  Msg.Cmd.Args.Common.SsidLen = (_u8)NameLen;
96 
97  /* Profile with no security */
98  /* Enterprise security profile */
99  if (NULL != pSecExtParams)
100  {
101  /* Update command opcode */
102  CmdCtrl.Opcode = SL_OPCODE_WLAN_WLANCONNECTEAPCOMMAND;
103  CmdCtrl.TxDescLen += sizeof(_WlanConnectEapCommand_t);
104  /* copy SSID */
105  sl_Memcpy(EAP_SSID_STRING(&Msg), pName, NameLen);
106  CmdCtrl.TxDescLen += NameLen;
107  /* Copy password if supplied */
108  if ((NULL != pSecParams) && (pSecParams->KeyLen > 0))
109  {
110  /* update security type */
111  Msg.Cmd.Args.Common.SecType = pSecParams->Type;
112  /* verify key length */
113  if (pSecParams->KeyLen > MAX_KEY_LEN)
114  {
115  return SL_INVALPARAM;
116  }
117  /* update key length */
118  Msg.Cmd.Args.Common.PasswordLen = pSecParams->KeyLen;
119  ARG_CHECK_PTR(pSecParams->Key);
120  /* copy key */
121  sl_Memcpy(EAP_PASSWORD_STRING(&Msg), pSecParams->Key, pSecParams->KeyLen);
122  CmdCtrl.TxDescLen += pSecParams->KeyLen;
123  }
124  else
125  {
126  Msg.Cmd.Args.Common.PasswordLen = 0;
127  }
128 
129  ARG_CHECK_PTR(pSecExtParams);
130  /* Update Eap bitmask */
131  Msg.Cmd.Args.EapBitmask = pSecExtParams->EapMethod;
132  /* Update Certificate file ID index - currently not supported */
133  Msg.Cmd.Args.CertIndex = pSecExtParams->CertIndex;
134  /* verify user length */
135  if (pSecExtParams->UserLen > MAX_USER_LEN)
136  {
137  return SL_INVALPARAM;
138  }
139  Msg.Cmd.Args.UserLen = pSecExtParams->UserLen;
140  /* copy user name (identity) */
141  if(pSecExtParams->UserLen > 0)
142  {
143  sl_Memcpy(EAP_USER_STRING(&Msg), pSecExtParams->User, pSecExtParams->UserLen);
144  CmdCtrl.TxDescLen += pSecExtParams->UserLen;
145  }
146  /* verify Anonymous user length */
147  if (pSecExtParams->AnonUserLen > MAX_ANON_USER_LEN)
148  {
149  return SL_INVALPARAM;
150  }
151  Msg.Cmd.Args.AnonUserLen = pSecExtParams->AnonUserLen;
152  /* copy Anonymous user */
153  if(pSecExtParams->AnonUserLen > 0)
154  {
155  sl_Memcpy(EAP_ANON_USER_STRING(&Msg), pSecExtParams->AnonUser, pSecExtParams->AnonUserLen);
156  CmdCtrl.TxDescLen += pSecExtParams->AnonUserLen;
157  }
158 
159  }
160 
161  /* Regular or open security profile */
162  else
163  {
164  /* Update command opcode */
165  CmdCtrl.Opcode = SL_OPCODE_WLAN_WLANCONNECTCOMMAND;
166  CmdCtrl.TxDescLen += sizeof(_WlanConnectCommon_t);
167  /* copy SSID */
168  sl_Memcpy(SSID_STRING(&Msg), pName, NameLen);
169  CmdCtrl.TxDescLen += NameLen;
170  /* Copy password if supplied */
171  if( NULL != pSecParams )
172  {
173  /* update security type */
174  Msg.Cmd.Args.Common.SecType = pSecParams->Type;
175  /* verify key length is valid */
176  if (pSecParams->KeyLen > MAX_KEY_LEN)
177  {
178  return SL_INVALPARAM;
179  }
180  /* update key length */
181  Msg.Cmd.Args.Common.PasswordLen = pSecParams->KeyLen;
182  CmdCtrl.TxDescLen += pSecParams->KeyLen;
183  /* copy key (could be no key in case of WPS pin) */
184  if( NULL != pSecParams->Key )
185  {
186  sl_Memcpy(PASSWORD_STRING(&Msg), pSecParams->Key, pSecParams->KeyLen);
187  }
188  }
189  /* Profile with no security */
190  else
191  {
192  Msg.Cmd.Args.Common.PasswordLen = 0;
193  Msg.Cmd.Args.Common.SecType = SL_SEC_TYPE_OPEN;
194  }
195  }
196  /* If BSSID is not null, copy to buffer, otherwise set to 0 */
197  if(NULL != pMacAddr)
198  {
199  sl_Memcpy(Msg.Cmd.Args.Common.Bssid, pMacAddr, sizeof(Msg.Cmd.Args.Common.Bssid));
200  }
201  else
202  {
203  _SlDrvMemZero(Msg.Cmd.Args.Common.Bssid, (_u16)sizeof(Msg.Cmd.Args.Common.Bssid));
204  }
205 
206 
207  VERIFY_RET_OK ( _SlDrvCmdOp(&CmdCtrl, &Msg, NULL));
208 
209  return (_i16)Msg.Rsp.status;
210 }

305 
306  Msg.Cmd.PolicyType = Type;
307  Msg.Cmd.PolicyOption = Policy;
308  VERIFY_RET_OK(_SlDrvCmdOp((_SlCmdCtrl_t *)&_SlPolicyGetCmdCtrl, &Msg, &CmdExt));

312  {
313  *pValLen = Msg.Rsp.PolicyOptionLen;
314  return SL_ESMALLBUF;
315  }
316  else
317  {
318  /* no pointer valus, fill the results into _i8 */
319  if( 0 == CmdExt.ActualRxPayloadLen )
320  {
321  pVal[0] = Msg.Rsp.PolicyOption;
322  *pValLen = 1;
323  }
324  /* user buffer also holds a place for the current policy */
325  else if ((*pValLen > 0) && (*pValLen > CmdExt.ActualRxPayloadLen))
326  {
327  pVal[*pValLen - sizeof(_u32)] = (_u32)Msg.Rsp.PolicyOption; /* the last _u32 in the user buffer holds the policy */
328  *pValLen = (_u8)CmdExt.ActualRxPayloadLen + 4;
329  }
330  else
331  {
332  *pValLen = CmdExt.ActualRxPayloadLen;
333  }
334 
335  }
336  return (_i16)SL_OS_RET_CODE_OK;
