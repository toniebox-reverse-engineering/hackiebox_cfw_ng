287 /******************************************************************************
288 
289  sl_NetAppMDNSRegisterService
290 
291  CALLER user from its host
292 
293 
294  DESCRIPTION:
295  Add/delete service
296  The function manipulates the command that register the service and call
297  to the NWP in order to add/delete the service to/from the mDNS package and to/from the DB.
298 
299  This register service is a service offered by the application.
300  This unregister service is a service offered by the application before.
301 
302  The service name should be full service name according to RFC
303  of the DNS-SD - means the value in name field in SRV answer.
304 
305  Example for service name:
306  1. PC1._ipp._tcp.local
307  2. PC2_server._ftp._tcp.local
308 
309  If the option is_unique is set, mDNS probes the service name to make sure
310  it is unique before starting to announce the service on the network.
311  Instance is the instance portion of the service name.
312 
313 
314 
315 
316  PARAMETERS:
317 
318  The command is from constant parameters and variables parameters.
319 
320  Constant parameters are:
321 
322  ServiceLen - The length of the service.
323  TextLen - The length of the service should be smaller than 64.
324  port - The port on this target host.
325  TTL - The TTL of the service
326  Options - bitwise parameters:
327  bit 0 - is unique (means if the service needs to be unique)
328  bit 31 - for internal use if the service should be added or deleted (set means ADD).
329  bit 1-30 for future.
330 
331  The variables parameters are:
332 
333  Service name(full service name) - The service name.
334  Example for service name:
335  1. PC1._ipp._tcp.local
336  2. PC2_server._ftp._tcp.local
337 
338  Text - The description of the service.
339  should be as mentioned in the RFC
340  (according to type of the service IPP,FTP...)
341 
342  NOTE - pay attention
343 
344  1. Temporary - there is an allocation on stack of internal buffer.
345  Its size is NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH.
346  It means that the sum of the text length and service name length cannot be bigger than
347  NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH.
348  If it is - An error is returned.
349 
350  2. According to now from certain constraints the variables parameters are set in the
351  attribute part (contain constant parameters)
352 
353 
354 
355  RETURNS: Status - the immediate response of the command status.
356  0 means success.
357 
358 
359 
360 ******************************************************************************/
361 _i16 sl_NetAppMDNSRegisterUnregisterService( const _i8* pServiceName,
362  const _u8 ServiceNameLen,
363  const _i8* pText,
364  const _u8 TextLen,
365  const _u16 Port,
366  const _u32 TTL,
367  const _u32 Options)
368 
369 {
370  _SlNetappMdnsRegisterServiceMsg_u Msg;
371  _SlCmdExt_t CmdExt ;
372  _i8 ServiceNameAndTextBuffer[NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH];
373  _i8 *TextPtr;
374 
375  /* verify no erorr handling in progress. if in progress than
376  ignore the API execution and return immediately with an error */
377  VERIFY_NO_ERROR_HANDLING_IN_PROGRESS();
378  /*
379 
380  NOTE - pay attention
381 
382  1. Temporary - there is an allocation on stack of internal buffer.
383  Its size is NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH.
384  It means that the sum of the text length and service name length cannot be bigger than
385  NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH.
386  If it is - An error is returned.
387 
388  2. According to now from certain constraints the variables parameters are set in the
389  attribute part (contain constant parameters)
390 
391 
392  */
393 
394  /*build the attribute part of the command.
395  It contains the constant parameters of the command*/
396 
397  Msg.Cmd.ServiceNameLen = ServiceNameLen;
398  Msg.Cmd.Options = Options;
399  Msg.Cmd.Port = Port;
400  Msg.Cmd.TextLen = TextLen;
401  Msg.Cmd.TTL = TTL;
402 
403  /*Build the payload part of the command
404  Copy the service name and text to one buffer.
405  NOTE - pay attention
406  The size of the service length + the text length should be smaller than 255,
407  Until the simplelink drive supports to variable length through SPI command. */
408  if(TextLen + ServiceNameLen > (NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH - 1 )) /*-1 is for giving a place to set null termination at the end of the text*/
409  {
410  return -1;
411  }
412 
413  _SlDrvMemZero(ServiceNameAndTextBuffer, (_u16)NETAPP_MDNS_MAX_SERVICE_NAME_AND_TEXT_LENGTH);
414 
415 
416  /*Copy the service name*/
417  sl_Memcpy(ServiceNameAndTextBuffer,
418  pServiceName,
419  ServiceNameLen);
420 
421  if(TextLen > 0 )
422  {
423 
424  TextPtr = &ServiceNameAndTextBuffer[ServiceNameLen];
425  /*Copy the text just after the service name*/
426  sl_Memcpy(TextPtr,
427  pText,
428  TextLen);
429 
430 
431  }
432 
433  _SlDrvResetCmdExt(&CmdExt);
434  CmdExt.TxPayloadLen = (TextLen + ServiceNameLen);
435  CmdExt.pTxPayload = (_u8 *)ServiceNameAndTextBuffer;
436 
437 
438  VERIFY_RET_OK(_SlDrvCmdOp((_SlCmdCtrl_t *)&_SlRegisterServiceCtrl, &Msg, &CmdExt));
439 
440  return (_i16)Msg.Rsp.status;

656 /*
657  Note:
658  1. The return's attributes are belonged to first service that is found.
659  It can be other services with the same service name will response to
660  the query. The results of these responses are saved in the peer cache of the NWP, and
661  should be read by another API.
662 
663  2. Text length can be 120 bytes only - not more
664  It is because of constraints in the NWP on the buffer that is allocated for the Async event.

687  This pointers are belonged to local struct that is set to global Async response parameter.
688  It is done in order not to run more than one sl_DnsGetHostByService at the same time.
689  The API should be run only if global parameter is pointed to NULL. */
690  AsyncRsp.out_pText = pText;
691  AsyncRsp.inout_TextLen = (_u16* )pTextLen;
692  AsyncRsp.out_pPort = pPort;
693  AsyncRsp.out_pAddr = (_u32 *)pAddr;
694 
695 
696  ObjIdx = _SlDrvProtectAsyncRespSetting((_u8*)&AsyncRsp, GETHOSYBYSERVICE_ID, SL_MAX_SOCKETS);
697 
698  if (MAX_CONCURRENT_ACTIONS == ObjIdx)
699  {
700  return SL_POOL_IS_EMPTY;
701  }
702 
703 
704  if (SL_AF_INET6 == Family)

707  }

710 
711 
712 
713  /* If the immediate reponse is O.K. than wait for aSYNC event response. */
714  if(SL_RET_CODE_OK == Msg.Rsp.status)
715  {
716  SL_DRV_SYNC_OBJ_WAIT_FOREVER(&g_pCB->ObjPool[ObjIdx].SyncObj);

738

742  1. IP of the service
743  2. The port of service.
744  3. The text of service.
745

747  It is similar to get host by name method.
748 
749  It is done by a single shot query with PTR type on the service name.
750 
751 
752 
753  Note:
754  1. The return's attributes are belonged to first service that is found.
755  It can be other services with the same service name will response to
756  the query. The results of these responses are saved in the peer cache of the NWP, and
757  should be read by another API.
758 
759 
760  PARAMETERS:
761 
762  pVoidBuf - is point to opcode of the event.
763  it contains the outputs that are given to the user
764 
765  outputs description:
766 
767  1.out_pAddr[] - output: Contain the IP address of the service.
768  2.out_pPort - output: Contain the port of the service.
769  3.inout_TextLen - Input: Contain the max length of the text that the user wants to get.
770  it means that if the test of service is bigger that its value than
771  the text is cut to inout_TextLen value.
772  Output: Contain the length of the text that is returned. Can be full text or part
773  of the text (see above).
774 
775  4.out_pText - Contain the text of the service (full or part see above- inout_TextLen description).
776
