97 _i16 sl_Start(const void* pIfHdl, _i8* pDevName, const P_INIT_CALLBACK pInitCallBack)
98 {
99  _u8 ObjIdx = MAX_CONCURRENT_ACTIONS;
100  InitComplete_t AsyncRsp;
101 
102  if ( g_bDeviceRestartIsRequired == (_u8)TRUE)
103  {
104  return SL_API_ABORTED;
105  }
106 
107  /* Perform any preprocessing before enable networking services */
108 #ifdef sl_DeviceEnablePreamble
109  sl_DeviceEnablePreamble();
110 #endif
111 
112  /* ControlBlock init */
113  (void)_SlDrvDriverCBInit();
114 
115  /* open the interface: usually SPI or UART */
116  if (NULL == pIfHdl)
117  {
118  g_pCB->FD = sl_IfOpen((void *)pDevName, 0);
119  }
120  else
121  {
122  g_pCB->FD = (_SlFd_t)pIfHdl;
123  }
124 
125  ObjIdx = _SlDrvProtectAsyncRespSetting((_u8 *)&AsyncRsp, START_STOP_ID, SL_MAX_SOCKETS);
126 
127  if (MAX_CONCURRENT_ACTIONS == ObjIdx)
128  {
129  return SL_POOL_IS_EMPTY;
130  }
131 
132  if( g_pCB->FD >= (_SlFd_t)0)
133  {
134  sl_DeviceDisable();
135 
136  sl_IfRegIntHdlr((SL_P_EVENT_HANDLER)_SlDrvRxIrqHandler, NULL);
137 
138  g_pCB->pInitCallback = pInitCallBack;
139 
140  sl_DeviceEnable();
141 
142  if (NULL == pInitCallBack)
143  {
144 #ifdef SL_TINY_EXT
145  _SlDrvSyncObjWaitForever(&g_pCB->ObjPool[ObjIdx].SyncObj);
146 #else
147  _SlReturnVal_t retVal;
148  retVal = _SlDrvSyncObjWaitTimeout(&g_pCB->ObjPool[ObjIdx].SyncObj,
149  INIT_COMPLETE_TIMEOUT,
150  SL_DRIVER_API_DEVICE_SL_START);
151  if (retVal)
152  {
153  return SL_API_ABORTED;
154  }
155 #endif
156  SL_SET_DEVICE_STARTED;
157 
158  /* release Pool Object */
159  _SlDrvReleasePoolObj(g_pCB->FunctionParams.AsyncExt.ActionIndex);
160  return _sl_GetStartResponseConvert(AsyncRsp.Status);
161  }
162  else
163  {
164  return SL_RET_CODE_OK;
165  }
166  }
167  return SL_BAD_INTERFACE;
168 }

171 /***************************************************************************
172 _sl_HandleAsync_InitComplete - handles init complete signalling to
173 a waiting object
174 ****************************************************************************/
175 _SlReturnVal_t _sl_HandleAsync_InitComplete(void *pVoidBuf)
176 {
177  InitComplete_t *pMsgArgs = (InitComplete_t *)_SL_RESP_ARGS_START(pVoidBuf);
178 
179  SL_DRV_PROTECTION_OBJ_LOCK_FOREVER();
180 
181  if(g_pCB->pInitCallback)
182  {
183  g_pCB->pInitCallback(_sl_GetStartResponseConvert(pMsgArgs->Status));
184  }
185  else
186  {
187  sl_Memcpy(g_pCB->ObjPool[g_pCB->FunctionParams.AsyncExt.ActionIndex].pRespArgs, pMsgArgs, sizeof(InitComplete_t));
188  SL_DRV_SYNC_OBJ_SIGNAL(&g_pCB->ObjPool[g_pCB->FunctionParams.AsyncExt.ActionIndex].SyncObj);
189  }
190 
191  SL_DRV_PROTECTION_OBJ_UNLOCK();
192 
193  if(g_pCB->pInitCallback)
194  {
195  SL_SET_DEVICE_STARTED;
196  _SlDrvReleasePoolObj(g_pCB->FunctionParams.AsyncExt.ActionIndex);
197  }
198 
199 
200  return SL_RET_CODE_OK;
201 }

297

569 /******************************************************************************
570 _SlDrvDeviceEventHandler - handles internally device async events
571 ******************************************************************************/
572 _SlReturnVal_t _SlDrvDeviceEventHandler(void* pEventInfo)
573 {
574  DeviceEventInfo_t* pInfo = (DeviceEventInfo_t*)pEventInfo;
575  _SlResponseHeader_t* pHdr = (_SlResponseHeader_t *)pInfo->pAsyncMsgBuff;
576  _BasicResponse_t *pMsgArgs = (_BasicResponse_t *)_SL_RESP_ARGS_START(pHdr);
577 
578  switch(pHdr->GenHeader.Opcode)
579  {
580  case SL_OPCODE_DEVICE_INITCOMPLETE:
581  _sl_HandleAsync_InitComplete(pHdr);
582  break;
583  case SL_OPCODE_DEVICE_STOP_ASYNC_RESPONSE:
584  _sl_HandleAsync_Stop(pHdr);
585  break;
586 
587 
588  case SL_OPCODE_DEVICE_ABORT:
589  {
590  /* release global lock of cmd context */
591  if (pInfo->bInCmdContext == TRUE)
592  {
593  SL_DRV_LOCK_GLOBAL_UNLOCK();
594  }
595 
596 #ifndef SL_TINY_EXT
597  _SlDriverHandleError(SL_DEVICE_ABORT_ERROR_EVENT,
598  *((_u32*)pMsgArgs), /* Abort type */
599  *((_u32*)pMsgArgs + 1)); /* Abort data */
600 #endif
601  }
602  break;
603 
604  case SL_OPCODE_DEVICE_ASYNC_GENERAL_ERROR:
605  {
606  SlDeviceEvent_t devHandler;
607  devHandler.Event = SL_DEVICE_GENERAL_ERROR_EVENT;
608  devHandler.EventData.deviceEvent.status = pMsgArgs->status & 0xFF;
609  devHandler.EventData.deviceEvent.sender = (SlErrorSender_e)((pMsgArgs->status >> 8) & 0xFF);
610  _SlDrvHandleGeneralEvents(&devHandler);
611  }
612 
613  break;
614  default:
615  SL_ERROR_TRACE2(MSG_306, "ASSERT: _SlDrvDeviceEventHandler : invalid opcode = 0x%x = %1", pHdr->GenHeader.Opcode, pHdr->GenHeader.Opcode);
616  }
617 
618  return SL_OS_RET_CODE_OK;
619 }
