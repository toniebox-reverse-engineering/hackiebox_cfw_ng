305 typedef struct
306 {
307  /*The below parameters are used to configure the advertise times and interval
308  For example:
309  If:
310  Period is set to T
311  Repetitions are set to P
312  Telescopic factor is K=2
313  The transmission shall be:
314  advertise P times
315  wait T
316  advertise P times
317  wait 4 * T
318  advertise P time
319  wait 16 * T ... (till max time reached / configuration changed / query issued)
320  */
321  _u32 t; /* Number of ticks for the initial period. Default is 100 ticks for 1 second. */
322  _u32 p; /* Number of repetitions. Default value is 1 */
323  _u32 k; /* Telescopic factor. Default value is 2. */
